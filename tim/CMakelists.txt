cmake_minimum_required(VERSION 3.0)

set(CMAKE_C_STANDARD 11)

set(CMAKE_C_FLAGS_DEBUG "${CMAKE_C_FLAGS_DEBUG}")

project(tim VERSION 0.1.0 LANGUAGES C)

set(CURSES_NEED_CURSES TRUE)
set(CURSES_NEED_WIDE TRUE)
set(CURSES_LIBRARY ${PDCURSES_ROOT}/wincon/pdcurses.lib)
set(CURSES_INCLUDE_PATH ${PDCURSES_ROOT})

find_package( Curses REQUIRED )
include_directories( ${CURSES_INCLUDE_DIRS} )

set(EXECUTABLE_OUTPUT_PATH ../${PROJECT_NAME}/${CMAKE_BUILD_TYPE})

add_executable(${PROJECT_NAME} tim.c)

target_link_libraries( ${PROJECT_NAME} ${CURSES_LIBRARIES} )
add_dependencies(${PROJECT_NAME} build_pdcurses)
